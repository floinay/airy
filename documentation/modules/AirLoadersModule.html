import {Injector, NgModule} from '@angular/core';

import {AIR_LOADER_DEFAULT_PROVIDERS} from './air-loaders.providers';

  declarations,
  exports: declarations,
  providers: AIR_LOADER_DEFAULT_PROVIDERS,
  imports: [
    AirIconModule,
    AirCdkModule

    LoadersStaticInjector.set(injector);
  }
}
